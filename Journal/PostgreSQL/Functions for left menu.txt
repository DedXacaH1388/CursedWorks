DROP FUNCTION IF EXISTS get_classes, get_subjects;


-- get class info for left menu
CREATE FUNCTION get_classes (_teacher_id integer)
RETURNS TABLE (
	class_id	integer,
	class_name	VARCHAR
)
AS $$
BEGIN
	RETURN QUERY
	SELECT classes.class_id, classes.class_name 
	FROM classes
	WHERE classes.class_id IN (	SELECT subjects.class_id
					   			FROM subjects
					   			WHERE subjects.teacher_id = _teacher_id
							  );
END;
$$ LANGUAGE 'plpgsql';


-- get subject info for left menu
CREATE FUNCTION get_subjects (_class_id integer)
RETURNS TABLE (
	subject_id		integer,
	subject_name	VARCHAR
)
AS $$
BEGIN
	RETURN QUERY
	SELECT subjects.subject_id, subjects.subject_name
	FROM subjects 
	WHERE subjects.class_id = _class_id;
END;
$$ LANGUAGE 'plpgsql';
